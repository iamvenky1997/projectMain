Fabnocci series using recursion : 

inp  = int(input("Enter a number : "))

def fabnocci(inp):
    if inp <= 1:
        return inp
    else:
        return fabnocci(inp-1)+fabnocci(inp-2)
        
if inp <= 0:
    print("Enter a positive Number")
else:
    print("Fabnocci series : ")
    for i in range(inp):
        print(fabnocci(i))
