Sum of Numbers in a digit

n = int(input("Enter Number : ")) //input from the user
n = 1234567 
sum = 0
while n >0:
    lastNumber = n % 10
    sum += lastNumber
    n = n//10
print(sum)

Sum of Numbers in a String

n = input() //input from the user
n = "1234567"
sum = 0
for i in n:
    sum += int(i)
print(sum)
